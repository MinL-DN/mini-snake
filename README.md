## 源码目录介绍
```

├── component                    // 各实体
│   ├── camera.js                // 摄像机类
│   ├── circle.js                // 离屏圆
│   ├── food-canvas.js           // 离屏食物层
│   └── snake.js                 // 蛇类
├── public                       // 静态资源存放
├── scene                        // 场景
│   ├── base.js                  // 场景基础方法
│   └── main.js                  // 主游戏场景
├── utils                        // 工具类
│   ├── canvas.js                // 画布生成工具
│   ├── dom.js                   // 定义游戏开发基础类
│   ├── gradientColor.js         // 渐变色工具（暂未使用）
│   ├── index.js                 // 工具类入口
│   ├── resource-ctrl.js         // 资源获取器
│   └── runtime.js               // ES6 运行块
└── game.js                      // 启动文件

```
